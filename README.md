# Yelp API Keys:

CONSUMER_SECRET = 'xxxxxxxxxxxxxxxxxxxxxx'
TOKEN = 'xxxxxxxxxxxxxxxxxxxxxxx'
TOKEN_SECRET = 'xxxxxxxxxxxxxxxxxxx'

# Settings
